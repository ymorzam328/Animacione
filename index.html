function sorteo() {
const num = [];
const primitiva = [];
for (let i = 1; i <= 49; i++) num.push(i);
for (let i = 0; i < 7; i++) {
let indice = Math.floor(Math.random() * num.length);
let number = num[indice];
num.splice(indice, 1);
primitiva.push(number);
if (i == 5) primitiva.sort((a, b) => a - b);
}
return primitiva;
};
window.addEventListener('DOMContentLoaded', () => {
const nums = sorteo();
document.querySelector(".nums").querySelectorAll("div").forEach((v, k) => v.textContent = nums[k]);
document.getElementById('fecha').textContent += new Date().toLocaleDateString();
});
</script>
